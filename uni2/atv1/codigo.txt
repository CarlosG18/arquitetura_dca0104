.data
    # vetor v - sinais obtidos entre 0 a 9 
    v: .word 9, 5, 7, 5, 3, 4, 0, 2, 6, 4, 2, 5, 4, 1, 2, 1, 6, 2, 2, 3, 6, 3, 0, 0, 7, 8, 3, 4, 5, 4, 0, 5, 2, 9, 8, 7
    # definindo o tamanho maximo de intensidade
    LMax: .word 9
    # definido o tamanho de elementos do vetor
    TAM: .word 36
    # definindo o vetor do histograma H, alocado com 9 espaços de 4 bytes cada = 36
    H: .space 36

.text
    main:

    # inicializando o vetor H com o valor 0 para cada indicie
    
    la $t0, H # obtendo o endereço base do vetor H e colocando no registrador temporario t0
    
    li $t1, 0 # alocando o valor 0 no registrador temporario t1
    
    la $t2, LMax # obtendo o valor de LMax e alocando no registrador temporario t2 

    loop:
        sw $t1, 0($t1) # armazena o valor zero no endereço atual do vetor H
        addiu $t0, $t0, 4 # incremento de 4 bytes no endereço de t0, ou seja, obtendo o endereço do proximo valor de H
        addiu $t2, $t2, -1 # decrementa em 1 o valor de LMax
        bnez $t2, loop # verifica se o valor guardado em t2 é diferente de zero, se o valor for false ele irá fazer um salto para o loop, caso contrario ele continuará o fluxo do codigo
    

    # realizando os devidos incrementos no vetor do histograma H
    la $s0, v # atribuindo o endereço base de v para o registrador S0
    la $s1, H # atribuindo o endereço base de H para o registrador S1
    la $t3, LMax # obtendo o valor de LMax e alocando no registrador temporario t3
    la $t4, TAM # obtendo o valor de TAM e alocando no registrador temporario t4

    loop:
        

        addiu $t4, $t4, -1 # decrementa o valor de TAM e reatribui ao registrador temporario t4
        bltz $t4, loop # enquanto o valor armazenado no registrador t4 for menor do que zero ele irá saltar para o comando loop, caso contrario, ele seguirá o fluxo do codigo
